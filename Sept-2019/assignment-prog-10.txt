int maxsubarray(int a[],int n)
{
int min=a[0],i;
for(i=0;i<n;i++)
{
if(a[i]<=min)
min=a[i];
}
int msf=min,meh=0;
int start=0,end=0,s=0;
for(int i=0;i<n;i++)
{
meh+=a[i];
if(msf<meh)
{
msf=meh;
start=s;
end=i;
}
if(meh<0)
{
meh=i+1;
s=i+1;
}
}
printf("\n sum is %d:",msf);
printf("[");
for(i=start;i<=end;i++)
printf("%d ",a[i]);
printf("]");
}
